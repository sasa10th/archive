§2. 그 외의 경우

① $R_i < 0$만 존재 - $M_i+R_i$ 기준 내림차순이 맞는지 검증
② $R_i < 0$만 존재 - $M_i+R_i$ 오름차순(일부러 틀린 정렬 유도)
③ 혼합: $R_i >= 0$ 그룹과 $R_i < 0$ 그룹이 섞인 경우
④ 혼합: $R_i < 0$인 문제의 $M_i$가 더 작아서 앞에 오면 안 되는 케이스
⑤ 극단값: $M_i$ 최대, $R_i$ 최소
⑥ $M_i$는 같고 $R_i$만 다름(타이브레이킹 검증)
⑦ $R_i$가 0인 경우만 존재
⑧ 랜덤 혼합(전체적인 정확성 검증)

                                    import random

                                    def save_case(filename, n, data):
                                        with open(filename, "w") as f:
                                            f.write(f"{n}\n")
                                            for m, r in data:
                                                f.write(f"{m} {r}\n")
                                        print(f"{filename} 생성 완료")

                                    N = 5000

                                    # R_i < 0 만 존재 - M_i+R_i 기준 내림차순이 맞는지 검증
                                    def gen_all_negative(n):
                                        return [(i + 1, -1) for i in range(1, n + 1)]

                                    # R_i < 0 만 존재 - M_i+R_i 오름차순(일부러 틀린 정렬 유도)
                                    def gen_all_negative_desc(n):
                                        return [(i + 1, -1) for i in range(n, 0, -1)]

                                    # 혼합: R_i >= 0 그룹과 R_i < 0 그룹이 섞인 경우
                                    def gen_mixed(n):
                                        return [(i + 1, 1 if i % 2 == 0 else -1) for i in range(1, n + 1)]

                                    # 혼합: R_i < 0인 문제의 M_i가 더 작아서 앞에 오면 안 되는 케이스
                                    def gen_mixed_trap(n):
                                        pos = [(10, 5)] * (n // 2)        # R_i > 0, M_i 작음
                                        neg = [(1, -1)] * (n - n // 2)    # R_i < 0, M_i 더 작음 → 앞에 두면 손해
                                        return pos + neg

                                    # 극단값: M_i 최대, R_i 최소
                                    def gen_extreme(n):
                                        return [(10**9, -10**9)] * n

                                    # M_i는 같고 R_i만 다름(타이브레이킹 검증)
                                    def gen_same_m_diff_r(n):
                                        return [(100, i - n // 2) for i in range(n)]

                                    # R_i가 0인 경우만 존재
                                    def gen_zero_r(n):
                                        return [(i + 1, 0) for i in range(1, n + 1)]

                                    # 랜덤 혼합(전체적인 정확성 검증)
                                    def gen_random_full(n):
                                        return [(random.randint(1, 10**9), random.randint(-10**9, 10**9)) for _ in range(n)]

                                    if __name__ == "__main__":
                                        save_case("2_1.in", N, gen_all_negative(N))
                                        save_case("2_2.in", N, gen_all_negative_desc(N))
                                        save_case("2_3.in", N, gen_mixed(N))
                                        save_case("2_4.in", N, gen_mixed_trap(N))
                                        save_case("2_5.in", N, gen_extreme(N))
                                        save_case("2_6.in", N, gen_same_m_diff_r(N))
                                        save_case("2_7.in", N, gen_zero_r(N))
                                        save_case("2_8.in", N, gen_random_full(N))
                                

                                    import os

                                    def solve(n, problems):
                                        # R >= 0: M_i 오름차순, R < 0: M_i + R_i 내림차순
                                        pos = sorted([p for p in problems if p[1] >= 0], key=lambda x: x[0])
                                        neg = sorted([p for p in problems if p[1] < 0], key=lambda x: x[0] + x[1], reverse=True)
                                        order = pos + neg

                                        min_s = 0
                                        prefix_r = 0
                                        for m, r in order:
                                            # 이 문제를 풀기 전 멘탈: S + prefix_r >= m 이어야 함
                                            min_s = max(min_s, m - prefix_r)
                                            prefix_r += r
                                            # 풀고 난 후 멘탈: S + prefix_r >= 0 이어야 함
                                            min_s = max(min_s, -prefix_r)

                                        return min_s

                                    def process(in_file, out_file):
                                        with open(in_file) as f:
                                            n = int(f.readline())
                                            problems = [tuple(map(int, f.readline().split())) for _ in range(n)]
                                        ans = solve(n, problems)
                                        with open(out_file, "w") as f:
                                            f.write(f"{ans}\n")
                                        print(f"{out_file} 생성 완료(답: {ans})")

                                    if __name__ == "__main__":
                                        for fn in sorted(os.listdir(".")):
                                            if fn.endswith(".in"):
                                                out_fn = fn.replace(".in", ".out")
                                                process(fn, out_fn)
